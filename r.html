     isPrime <- function(num){
    if(num <= 1){
        return(FALSE)
    }
    else if(num == 2){
        return(TRUE)
    }
    for(i in 2:sqrt(num)){
        if(num %% i == 0){
            return(FALSE)
        }
    }
    return(TRUE)
}

n <- 100
primenum <- c() #creating a empty arry so as to print the output one after the other # nolint

    if(isPrime(i)){
        primenum <- c(primenum,i) #adding the prime numbers one by one to the array 
    } #else we can print the output as print(i) this gives the output to next line one after the other
}
print(primenum) #printing the array

quad <- function(a, b, c) {
    if ((a * b * c) == 0) {
        print("Roots cannot be calculated")
        q()
    }

    d <- (b^2) - (4 * a * c)

    if (d == 0) {
        print("Roots are Real & Equal")
        x1 <- x2 <- (-b) / (2 * a)
        print(paste("X1:", x1))
        print(paste("X2:", x2))
    } else if (d > 0) {
        print("Roots are Real & Distinct")
        x1 <- (-b + sqrt(d)) / (2 * a)
        x2 <- (-b - sqrt(d)) / (2 * a)
        print(paste("X1:", x1))
        print(paste("X2:", x2))
    } else {
        print("Roots are Complex & Imaginary")
        realPart <- (-b) / (2 * a)
        imagPart <- sqrt(abs(d)) / (2 * a)
        root1 <- complex(real = realPart, imaginary = imagPart)
        root2 <- complex(real = realPart, imaginary = -imagPart)
        print(paste("X1: ",root1))
        print(paste("X2:", root2))
    }
}
quad(x, y, z)

fib <- function(n){
    if(n == 0){
    return(0)
    }
    else if (n == 1) {
    return(1)
    }
    else{
    return(fib(n-1) + fib(n-2))
    }
}

    print(fib(i),end=" ")
}

flag <- function(age, tshirt_size) {
    if (age <= 20 && (tshirt_size == "S" || tshirt_size == "M")) {
        return("Yes")
    } else if ((age > 20) && (tshirt_size == "L" || tshirt_size == "XL" || tshirt_size == "XXL")) {
        return("Yes")
    } else {
        return("No")
    }
}

flags <- mapply(flag, age, tshirt_size)

data <- data.frame(
    Participant = name,
    Age = age,
    Tshirt_Size = tshirt_size,
    Flag = flags
)

print(data)

library(titanic)

# Load the Titanic dataset
data(titanic_train)

# Calculate survivors by gender
survivors_gender <- table(titanic_train$Survived, titanic_train$Sex)

# Create a pie chart for survivors by gender
pie(survivors_gender, main = "Survivors by Gender",
    labels = c("Female Died", "Female Survived", "Male Died", "Male Survived"),
    col = c("red", "green", "blue", "yellow"),
    explode = c(0, 0.1, 0, 0.1),
    font.main = 1)

# Create a histogram for survivors by age group
hist(titanic_train$Age[titanic_train$Survived == 1], breaks = seq(0, 80, by = 5),
     col = "green", main = "Survivors by Age Group", xlab = "Age Groups")

# Calculate deaths by gender
death_gender <- table(!titanic_train$Survived, titanic_train$Sex)

# Create a barplot for deaths by gender
barplot(death_gender, main = "Deaths by Gender",
        xlab = "Gender", ylab = "Number of Deaths",
        names.arg = c("Female", "Male"),
        col = c("red", "blue"),
        legend = c("Died", "Survived"))

# Calculate survival rate by passenger class
survival_class <- aggregate(Survived ~ Pclass, data = titanic_train, FUN = mean)

# Create a line plot for survival rate by passenger class
plot(survival_class$Pclass, survival_class$Survived, type = "b",
     main = "Survival Rate by Passenger Class",
     xlab = "Passenger Class", ylab = "Survival Rate")
