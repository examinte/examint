#python program to convert binary to devimal, octal to hexadecimal using functions.
def bin_to_deci(n, b):
d = 0
p = 0
while n > 0:
d1 = n % 10
d = d + (d1 * (b ** p))
p = p + 1
n = n // 10
return d

def oct_to_hexa(n, b):
hexa = ""
decimal = bin_to_deci(n, 8) # Corrected function name here
hexachar = "0123456789ABCDEF"
while decimal > 0:

h = hexachar[d]

from difflib import SequenceMatcher
def similarity(str1,str2):
similarity_ratio = SequenceMatcher(None,str1,str2).ratio()
return similarity_ratio

#python program to implement insertion sort and merge sort using list
def insertion_sort(arr):
for i in range(1, len(arr)):
key = arr[i]
j = i - 1
while j >= 0 and key < arr[j]:
arr[j + 1] = arr[j]
j -= 1
arr[j + 1] = key
return arr

def merge_sort(arr):
if len(arr) <= 1:
return arr

mid = len(arr) // 2
left_half = arr[:mid]
right_half = arr[mid:]

left_half = merge_sort(left_half)
right_half = merge_sort(right_half)

result = []
left_index, right_index = 0, 0

while left_index < len(left_half) and right_index < len(right_half):
if left_half[left_index] < right_half[right_index]:
result += [left_half[left_index]]
left_index += 1
else:
result += [right_half[right_index]]
right_index += 1

while left_index < len(left_half):
result += [left_half[left_index]]
left_index += 1

while right_index < len(right_half):
result += [right_half[right_index]]
right_index += 1
return result


arr = [66,89,1,45,3,23,11]
print("Array Before Sorting :",arr)
print("Insertion Sort Result:",insertion_sort(arr) )
print("Merge Sort Result :",merge_sort(arr))

# Write a program to convert roman numbers to integer values using dictionaries.
def roman_to_int(roman):
roman_numerals = {
'I': 1, 'V': 5, 'X': 10, 'L': 50,
'C': 100, 'D': 500, 'M': 1000
}

total = 0
prev = 0

for char in roman:
value = roman_numerals[char]
if value > prev:
total += value - 2 * prev#total=total+value-3
else:
total += value
prev = value

return total


romanNumber = input("Enter a Roman Number: ")
integerValue = roman_to_int(romanNumber.upper())
print("Integer value:", integerValue)

def isPhoneNumber(text):
if len(text) != 12:
return False
for i in range(0, 3):
if not text[i].isnumeric():
return False
if text[3] != '-':
return False
for i in range(4, 7):
if not text[i].isnumeric():
return False
if text[7] != '-':
return False
for i in range(8, 12):
if not text[i].isnumeric():
return False
return True

import re

def find_patterns(pattern, text):
return re.findall(pattern, text, re.IGNORECASE)

file_path = "phone.txt"

try:
with open(file_path, 'r') as file:
text = file.read()

phone_numbers = find_patterns(r'\+\d{11}', text)
email_addresses = find_patterns(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,7}\b', text)

print("Phone numbers found:")
print('\n'.join(phone_numbers))

print("\nEmail addresses found:")
print('\n'.join(email_addresses))

except FileNotFoundError:
print("File not found!")
