#python program to convert binary to devimal, octal to hexadecimal using functions.
def bin_to_deci(n, b):
    d = 0
    p = 0
    while n > 0:
        d1 = n % 10
        d = d + (d1 * (b ** p))
        p = p + 1
        n = n // 10
    return d

def oct_to_hexa(n, b):
    hexa = ""
    decimal = bin_to_deci(n, 8)  # Corrected function name here
    hexachar = "0123456789ABCDEF"
    while decimal > 0:
        d = decimal % 16
        h = hexachar[d]
        decimal = decimal // 16
        hexa = h + hexa
    return hexa

#python program to implement insertion sort and merge sort using list
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

def merge_sort(arr):
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left_half = arr[:mid]
    right_half = arr[mid:]

    left_half = merge_sort(left_half)
    right_half = merge_sort(right_half)

    result = []
    left_index, right_index = 0, 0

    while left_index < len(left_half) and right_index < len(right_half):
        if left_half[left_index] < right_half[right_index]:
            result += [left_half[left_index]]
            left_index += 1
        else:
            result += [right_half[right_index]]
            right_index += 1

    while left_index < len(left_half):
        result += [left_half[left_index]]
        left_index += 1

    while right_index < len(right_half):
        result += [right_half[right_index]]
        right_index += 1
    return result


arr = [66,89,1,45,3,23,11]
print("Array Before Sorting :",arr)
print("Insertion Sort Result:",insertion_sort(arr) )
print("Merge Sort Result    :",merge_sort(arr))

# Write a program to convert roman numbers to integer values using dictionaries.
def roman_to_int(roman):
    roman_numerals = {
        'I': 1, 'V': 5, 'X': 10, 'L': 50,
        'C': 100, 'D': 500, 'M': 1000
    }

    total = 0
    prev = 0

    for char in roman:
        value = roman_numerals[char]
        if value > prev:
            total += value - 2 * prev#total=total+value-3
        else:
            total += value
        prev = value

    return total


romanNumber = input("Enter a Roman Number: ")
integerValue = roman_to_int(romanNumber.upper())
print("Integer value:", integerValue)

def isPhoneNumber(text):
    if len(text) != 12:
        return False
    for i in range(0, 3):
        if not text[i].isnumeric():
            return False
    if text[3] != '-':
        return False 
    for i in range(4, 7):
        if not text[i].isnumeric():
            return False
    if text[7] != '-':
        return False
    for i in range(8, 12):
        if not text[i].isnumeric():
            return False
    return True

#  python program to accept a file name from the user and perform the following

# 1. Display the first N line of the file
# 2. Find the frequency of occurrence of the word accepted from the user in the
# filed
def display_first_n_lines(file_name, n):
    try:
        with open(file_name, 'r') as file:
            for _ in range(n):
                line = file.readline()
                if not line:
                    break
                print(line, end='')
    except FileNotFoundError:
        print("File not found.")

def count_word_frequency(file_name, word):
    try:
        with open(file_name, 'r') as file:
            content = file.read()
            word_count = content.lower().count(word.lower())
            print(f"The word '{word}' appears {word_count} times in the file.")
    except FileNotFoundError:
        print("File not found.")

file_name = input("Enter the file name: ")

try:
    n = int(input("Enter the number of lines to display: "))
    display_first_n_lines(file_name, n)
    
    word_to_find = input("Enter the word to find frequency: ")
    count_word_frequency(file_name, word_to_find)
except ValueError:
    print("Invalid input. Please enter a valid number.")

import os
import zipfile

def create_zip(folder_path, zip_file_path):
    with zipfile.ZipFile(zip_file_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
        for root, _, files in os.walk(folder_path):
            for file in files:
                file_path = os.path.join(root, file)
                zipf.write(file_path, os.path.relpath(file_path, folder_path))

def extract_zip(zip_file_path, extract_folder_path):
    is_exist = os.path.exists(extract_folder_path)
    if not is_exist:
        os.makedirs(extract_folder_path)
        
    with zipfile.ZipFile(zip_file_path, 'r') as example_zip:
        example_zip.extractall(extract_folder_path)
        print("Files extracted successfully.")

# By using the concept of inheritance write a python program to find the area of triangle,
# circle and rectangle.
import math
class Shape:
    def area(self):
        pass

class Triangle(Shape):
    def area(self, base, height):
        return 0.5 * base * height

class Circle(Shape):
    def area(self, radius):
        return math.pi * radius ** 2

class Rectangle(Shape):
    def area(self, length, width):
        return length * width

triangle = Triangle()
circle = Circle()
rectangle = Rectangle()

print("Area of Triangle:", triangle.area(10, 5))
print("Area of Circle:", circle.area(7))
print("Area of Rectangle:", rectangle.area(8, 6))

# Write a python program to find the whether the given input is palindrome or not (for
# both string and integer) using the concept of polymorphism and inheritance.
class PalindromeChecker:
    def is_palindrome(self, value):
        pass

class StringPalindrome(PalindromeChecker):
    def is_palindrome(self, value):
        cleaned_value = value.lower().replace(" ", "")  
        return cleaned_value == cleaned_value[::-1]  

class IntegerPalindrome(PalindromeChecker):
    def is_palindrome(self, value):
        return str(value) == str(value)[::-1]  

def check_palindrome(palindrome_checker, value):
    if palindrome_checker.is_palindrome(value):
        return f"'{value}' is a palindrome."
    else:
        return f"'{value}' is not a palindrome."

input_string = input("Enter a string: ")
input_integer = int(input("Enter an integer: "))

string_checker = StringPalindrome()
integer_checker = IntegerPalindrome()

print(check_palindrome(string_checker, input_string))
print(check_palindrome(integer_checker, input_integer))

# Write a python program to download the all XKCD comics
import os
import requests
from bs4 import BeautifulSoup
import urllib

if not os.path.exists('xkcd_comics'):
    os.makedirs('xkcd_comics')

base_url = 'https://xkcd.com/'

response = requests.get(base_url)
soup = BeautifulSoup(response.text, 'html.parser')
latest_comic_num = int(soup.find('div', id='middleContainer').find('a', rel='prev')['href'][1:-1])

for comic_num in range(1, latest_comic_num + 1):
    url = base_url + str(comic_num)
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    img_url = soup.find('div', id='comic').find('img')['src']
    img_url = 'https:' + img_url if img_url.startswith('//') else img_url
    img_name = img_url.split('/')[-1]
    img_path = os.path.join('xkcd_comics', img_name)

    urllib.request.urlretrieve(img_url, img_path)
    print(f"Downloaded comic {comic_num}: {img_name}")
